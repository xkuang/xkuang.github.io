    for i in range(n):

    nums[i] = 1
    # Use the index as a hash key and the sign of the number to indicate presence
    for i in range(n):

    nums[0] = -abs(nums[0])

    nums[a] = -abs(nums[a])
    # Find the first missing positive integer
    for i in range(1, n):
    if nums[i] > 0:

    return n
    return n + 1
